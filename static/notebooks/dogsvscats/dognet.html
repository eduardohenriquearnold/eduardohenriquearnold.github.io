Create Model¶

In [81]:
from keras.models import Sequential
from keras.layers.core import Dense, Flatten, Dropout
from keras.layers.convolutional import Conv2D
from keras.layers.pooling import MaxPooling2D
from keras.layers.normalization import BatchNormalization

model = Sequential()
reg = keras.regularizers.l2(0.001)
model.add(BatchNormalization(input_shape=image_size+(3,)))
model.add(Conv2D(32, (3,3), activation='relu', kernel_regularizer=reg))
model.add(Conv2D(32, (3,3), activation='relu', kernel_regularizer=reg))
model.add(MaxPooling2D((2,2)))
model.add(Dropout(0.5))

model.add(Conv2D(16, (3,3), activation='relu', kernel_regularizer=reg))
model.add(Conv2D(16, (3,3), activation='relu', kernel_regularizer=reg))
model.add(MaxPooling2D((2,2)))
model.add(Dropout(0.5))

model.add(Conv2D(4, (3,3), activation='relu', kernel_regularizer=reg))
model.add(Conv2D(4, (3,3), activation='relu', kernel_regularizer=reg))
model.add(MaxPooling2D((2,2)))
model.add(Dropout(0.5))

model.add(Conv2D(1, (3,3), activation='relu', kernel_regularizer=reg))
model.add(Flatten())
model.add(Dense(2, activation='softmax', kernel_regularizer=reg))
