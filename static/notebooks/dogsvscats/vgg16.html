Create VGG Model with custom output layer¶

In [16]:
import vgg16
vgg = vgg16.VGG16(include_top=False, weights='imagenet', input_shape=(224,224,3))

#freeze convolutional layers
for layer in vgg.layers:
    layer.trainable = False
    
#add a FC classification layer with two output classes
x = keras.layers.core.Flatten()(vgg.output)
x = keras.layers.core.Dense(2, activation='softmax')(x)

model = keras.models.Model(vgg.input, x)
In [17]:

import csv
from keras.preprocessing.image import load_img, img_to_array

folder = 'data/eval_kagg/'

f = open('kaggle_submission.csv','w')
submission = csv.writer(f)
submission.writerow(['id','label'])

n=12500
for i in range(1,n+1):
    img = load_img(folder+str(i)+".jpg", target_size=(224,224,3))
    img = img_to_array(img)    
    img = np.expand_dims(img, axis=0)
    p_dog = model.predict(img)[0,1]
    submission.writerow([i, round(p_dog,3)])

f.close()
print("Done!")
